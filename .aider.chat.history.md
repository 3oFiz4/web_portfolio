// Minimalist button that opens a link in a new tab when clicked

/**
 * Creates a button element that opens the specified URL in a new browser tab.
 *
 * @param {string} text - The label to display on the button.
 * @param {string} url  - The URL to open when the button is clicked.
 * @returns {HTMLButtonElement} The created button element.
 */
function createOpenLinkButton(text, url) {
    const button = document.createElement('button');
    button.textContent = text || 'Open Link';

    // Simple styling to keep the button minimalist
    button.style.padding = '8px 12px';
    button.style.fontSize = '14px';
    button.style.cursor = 'pointer';
    button.style.border = '1px solid #ccc';
    button.style.backgroundColor = '#f9f9f9';
    button.style.borderRadius = '4px';

    // Open the URL in a new tab when clicked
    button.addEventListener('click', () => {
        window.open(url, '_blank');
    });

    return button;
}

// Example usage: automatically add the button to the page once the DOM is ready
document.addEventListener('DOMContentLoaded', () => {
    const link = 'https://example.com'; // Replace with your desired URL
    const btn = createOpenLinkButton('Go to Example', link);
    document.body.appendChild(btn);
});
